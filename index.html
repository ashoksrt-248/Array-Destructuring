Array destructuring In Javascript

var colors = ["Violet", "Indigo", "Blue", "Green", "Yellow", "Orange", "Red"];
// destructuring assignment
var[color1, color2, color3,color4,color5,color6,color7] = colors;
console.log(color1); // Violet
console.log(color2); // Indigo
console.log(color3); // Blue
console.log(color4); // Green
console.log(color5); // Yellow
console.log(color6); // Orange
console.log(color7); // Red

var lights = ["Violet", "Indigo", "Blue", "Green", "Yellow", "Orange", "Red"];
// destructuring assignment
var[light1, ,light3, ,light5] = lights; //Leave space for unpick elements
console.log(light1); // Violet
console.log(light3); // Blue
console.log(light5); // Yellow
console.log(light2); // Uncaught ReferenceError: light2 is not defined
console.log(light4); // Uncaught ReferenceError: light2 is not defined

Swapping Variables

var x = 100, y = 200;
[x, y] = [y, x];


console.log(x); // 200
console.log(y); // 100

Parsing returned array from functions

function array() {
return [100, 200, 300];
}
var [a, b, c] = array();
console.log(a); // 100
console.log(b); // 200
console.log(c); // 300
